```python
    async def search(

async def merge_results(
    providers: list[SearchProvider],
    query: str,
    target_date: date,
    max_results: int,
) -> list[NewsItem]:
    """并发调用多个 Provider，合并结果并去重。"""
    tasks = [p.search(query, target_date, max_results) for p in providers]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    merged: list[NewsItem] = []
    for r in results:
        if isinstance(r, list):
            merged.extend(r)
        elif isinstance(r, Exception):
            print(f"[warn] Provider failed: {r}")
    return dedupe(merged)[:max_results]
```

**要点**：
- `BochaProvider` 使用 `urllib` 还是 `aiohttp`，取决于是否引入 FastAPI 的 async 框架。
- 博查返回的结构通常包含 `title`、`url`、`snippet`、`summary`、`date` 字段，需要映射到现有的 `NewsItem`。
- 日期解析要兼容博查返回的格式（可能是 `"2026-06-03T10:00:00+08:00"` 这样的 ISO 格式）。
- `fresh` 参数用于限定时间窗口，MVP 可以设为 `oneDay`。

```kotlin
class BriefingRepository {
    private val api = ApiClient.service

    suspend fun loadBriefing(): Briefing {
        val response = api.getBriefing("cat_ai")
        return response.toDomain()
    }

    suspend fun refreshCategory(catId: String): String {

        return response.task_id

    suspend fun pollTaskStatus(taskId: String): String {
```

```kotlin
class BriefingViewModel : ViewModel() {
    private val repo = BriefingRepository()

    private val _briefing = MutableStateFlow<Briefing?>(null)
    val briefing: StateFlow<Briefing?> = _briefing

    private val _isLoading = MutableStateFlow(false)
    val isLoading: StateFlow<Boolean> = _isLoading

    private val _error = MutableStateFlow<String?>(null)
    val error: StateFlow<String?> = _error

    fun loadBriefing() {
        viewModelScope.launch {
            _isLoading.value = true
            _error.value = null
            try {
                _briefing.value = repo.loadBriefing()
            } catch (e: Exception) {
                _error.value = e.message ?: "加载失败"
            } finally {
                _isLoading.value = false
            }
        }
    }

    fun refreshCategory(catId: String) {
        viewModelScope.launch {
            _isLoading.value = true
            try {
                val taskId = repo.refreshCategory(catId)
                pollUntilDone(taskId)
                loadBriefing()  // 刷新完成后重新拉取
            } catch (e: Exception) {
                _error.value = e.message ?: "刷新失败"
            } finally {
                _isLoading.value = false
            }
        }
    }

    private suspend fun pollUntilDone(taskId: String) {
        var status = "pending"
        while (status in listOf("pending", "running")) {
            delay(2000)
            status = repo.pollTaskStatus(taskId)
        }
    }
}
```

**要点**：
- `ViewModel` 使用 `viewModelScope` 管理协程生命周期，避免内存泄漏。
- 下拉刷新触发 `refreshCategory` → 轮询任务状态 → 刷新简报数据。
- MockData 保留作为 fallback：API 不可用时显示 mock 数据。

**修改**：`backend/infopulse/api/main.py` 添加启动钩子：

```python
from contextlib import asynccontextmanager
from ..scheduler import scheduler, start_scheduler
from ..db import init_db

@asynccontextmanager
async def lifespan(app: FastAPI):
    await init_db()
    start_scheduler()
    yield
    scheduler.shutdown()

app = FastAPI(title="InfoPulse API", version="0.2.0", lifespan=lifespan)
```

**要点**：
- 使用 `APScheduler` 的异步调度器，与 FastAPI 事件循环共存。
- 每个分类独立生成：一个分类失败不影响其他分类。
- 生成结果自动写入数据库，无需手动触发。

## 6. 关键技术决策

### 6.1 为什么选 SQLite 而非 PostgreSQL（MVP）

| 考量 | SQLite | PostgreSQL |
| --- | --- | --- |
| 部署复杂度 | 零配置，单文件 | 需独立安装/配置 |
| 并发能力 | 单写多读（MVP 够用） | 高并发 |
| 切换时机 | — | 日活 >1000 或需多实例部署时 |

### 6.2 为什么搜索走 Provider 抽象

- 博查可能未来涨价/限流，有抽象层可以无缝切换为其他搜索源。
- 测试时可以 mock `SearchProvider`，无需真实 API Key。
- 后续可扩展：智谱 GLM web_search、百度千帆等。

### 6.3 为什么 Android 用 Retrofit 而非 Ktor

- MVP 阶段优先选纯 Android 方案，降低依赖复杂度。
- Retrofit 社区成熟、文档丰富、踩坑少。
- 后续若引入 KMP，再将网络层替换为 Ktor。

### 6.4 为什么不用 Redis

- MVP 阶段简报数量少（每天 3 条），SQLite 完全够用。
- 缓存、限流、去重都可在应用层内存实现。
- 正式部署后若需要分布式缓存/消息队列再引入 Redis。
